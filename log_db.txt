DROP TABLE `registration`.`users` IF EXISTS;
CREATE TABLE `registration`.`users` ( `id` INT UNSIGNED NOT NULL AUTO_INCREMENT , `login` VARCHAR(255) NOT NULL , `password` VARCHAR(255) NOT NULL , `remember_token` VARCHAR(255) NULL DEFAULT NULL , `email` VARCHAR(255) NOT NULL , `file_name` VARCHAR(255) NULL , `role_id` INT NOT NULL DEFAULT '0' , `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB   DEFAULT CHARSET=utf8 COLLATE=utf8_general_ci;
